t-SNE sample:

load mnist_train.mat
% train_X           60000x784
%  train_labels      60000x1
ind = randperm(size(train_X, 1));
train_X = train_X(ind(1:5000), :);
train_labels = train_labels(ind(1:5000), :);
no_dims = 2;
initial_dims = 50;
perplexity = 30;

mappedX = tsne(train_X, [], no_dims, initial_dims, perplexity);
h1=figure; h = gscatter(mappedX(:, 1), mappedX(:, 2), train_labels); saveas(h1, 'mnist.fig');
